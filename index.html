A sample API

						
harvester({
    adapter: 'mongodb',
    connectionString: "mongodb://localhost:27017/test"
    })
    .resource('artists', {
        name: Types.string().required().description('Artist name')
        links: {
        	genre: 'genres'
        }
    })
    .resource('genres', {
        name: Types.string()
    })
    .listen(1730);
